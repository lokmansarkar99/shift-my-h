```typescript
// jobStatusManager.ts
private jobs: Map<string, Job> = new Map();

createJob(jobData) {
  const newJob = { ...jobData, id, reference, stops, ... };
  this.jobs.set(newJob.id, newJob);  // ✅ Saved to Map
  this.saveJobToBackend(newJob);  // ✅ Send to backend
  this.emit('job_created', newJob);  // ✅ Notify listeners
  return newJob;
}
```
